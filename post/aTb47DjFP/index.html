题解：进制转换

2020-03-11 模拟 算法 数学

题目相关

题目描述：

请你编一程序实现两种不同进制之间的数据转换。

输入格式：

共三行，第一行是一个正整数，表示需要转换的数的进制n(2≤n≤16)，第二行是一个n进制数，若n>10则用大写字母A−F表示数码10-15，并且该n进制数对应的十进制的值不超过1000000000，第三行也是一个正整数，表示转换之后的数的进制(2≤m≤16)。

输出格式：

一个正整数，表示转换之后的m进制数。

样例：

input

16
FF
2

output

11111111

解题

关键前置知识

带*的为可选。

  • C++基础
    • 字符串及之前的知识点
    • *函数
  • 进制转换

分析

读完题目易发现，题目其实就是想要你将n进制的数字转换成对应的m进制的值。

直接硬转可能有点难，但我们可以中转一下，先将n进制的转换成十进制。再将十进制的值转换成m进制的值。

    int n,m;
    char num[15]={0},ten=0;
    cin>>n>>num>>m;
    // 1. 计算n进制数字num对应的十进制值，存放到ten中
    // 2. 计算十进制数字ten对应的m进制值，存放到num中
    // 3. 输出答案

n进制转十进制

先是要注意一点，此题最高可为十六进制，表示过程中存在字母，故使用字符串形式进行处理。

    // 1. 计算n进制数字num对应的十进制值，存放到ten中
    int len1=strlen(num);
    for(int i=len1-1,k=0;i>=0;i--,k++)
    {
        int tmp;
        if(num[i]>='A'&&num[i]<='F')
        {//A~F 分别转换为 10~15
            tmp=num[i]-'A'+10;
        }else
        {// 字符数字转换为整数数字
            tmp=num[i]-'0';
        }
        ten+=pow(tmp,k);
    }

十进制转m进制

// 2. 计算十进制数字ten对应的m进制值，存放到ans中
    int k=0;
    while(ten!=0)
    {
        int u=ten%m;
        ten/=m;
        char tmp;
        if(u>=10)
        {
            tmp=u+'A'-10;
        }else
        {
            tmp='0'+u;
        }
        ans[k++]=tmp;
    }

输出答案

完整代码

#include <iostream>
#include <cstring>
#include <cmath>
using namespace std;
int main()
{
    int n,m,ten=0;
    char num[10000]={0},ans[10000]={0};
    cin>>n>>num>>m;
    // 1. 计算n进制数字num对应的十进制值，存放到ten中
    int len1=strlen(num);
    for(int i=len1-1,k=0;i>=0;i--,k++)
    {
        int tmp;
        if(num[i]>='A'&&num[i]<='F')
        {//A~F 分别转换为 10~15
            tmp=num[i]-'A'+10;
        }else
        {// 字符数字转换为整数数字
            tmp=num[i]-'0';
        }
        
        ten=ten+tmp*pow(n,k);
    }
   // cout<<"ten:"<<ten<<endl;
    // 2. 计算十进制数字ten对应的m进制值，存放到ans中
    int k=0;
    while(ten!=0)
    {
        int u=ten%m;
        ten/=m;
        char tmp;
        if(u>=10)
        {
            tmp=u+'A'-10;
        }else
        {
            tmp='0'+u;
        }
        ans[k++]=tmp;
    }
    // 3. 输出答案
    for(int i=k-1;i>=0;i--)
    {
    	cout<<ans[i];
	}
    return 0;
}

可读性优化

可将中间进制转换的过程使用自定义函数来实现。

#include <iostream>
#include <cstring>
#include <cmath>
using namespace std;

//计算n进制数字num对应的十进制值
int nToTen(char num[],int n);
//计算十进制数字ten对应的m进制值，存放到ans中,返回长度
int tenToM(int ten,char ans[],int m);

int main()
{
    int n,m;
    char num[10000]={0},ans[10000]={0};
    cin>>n>>num>>m;
    // 1. 计算n进制数字num对应的十进制值，存放到ten中
    int ten=nToTen(num,n);
    // 2. 计算十进制数字ten对应的m进制值，存放到ans中
    int k=tenToM(ten,ans,m);
    // 3. 输出答案
    for(int i=k-1;i>=0;i--)
    {
    	cout<<ans[i];
	}
    return 0;
}

int nToTen(char num[],int n)
{//计算n进制数字num对应的十进制值
    int ten=0;
    int len1=strlen(num);
    for(int i=len1-1,k=0;i>=0;i--,k++)
    {
        int tmp;
        if(num[i]>='A'&&num[i]<='F')
        {//A~F 分别转换为 10~15
            tmp=num[i]-'A'+10;
        }else
        {// 字符数字转换为整数数字
            tmp=num[i]-'0';
        }
        
        ten=ten+tmp*pow(n,k);
    }
    return ten;
}
int tenToM(int ten,char ans[],int m)
{
    int k=0;
    while(ten!=0)
    {
        int u=ten%m;
        ten/=m;
        char tmp;
        if(u>=10)
        {
            tmp=u+'A'-10;
        }else
        {
            tmp='0'+u;
        }
        ans[k++]=tmp;
    }
    return k;
}
